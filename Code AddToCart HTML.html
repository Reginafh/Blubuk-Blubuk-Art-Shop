                cartItemsDiv.appendChild(div);
            });

            cartTotalSpan.textContent = total.toLocaleString('id-ID');
        }
    </script>

Karena kita menggunakan HTML statis (tanpa backend), maka kita akan menyimpan data produk di localStorage (penyimpanan browser lokal). Dengan cara ini:

  2. cart.html: membaca dari localStorage dan menampilkan daftar produk yang ditambahkan.

âœ… 1. produk.html (Halaman Produk Detail)

html
<!DOCTYPE html>
<html lang="id">
<head>
    <meta charset="UTF-8">
    <title>Detail Produk</title>
</head>
<body>
    <h1>Detail Produk</h1>

    <div class="product">
        <h2 id="product-name">Jam Tangan Sport</h2>

    <script>
        function addToCart() {
            // Data produk
            const name = document.getElementById('product-name').textContent;
            const price = parseInt(document.querySelector('#product-price span').textContent);

            const product = { name, price };

            // Ambil cart dari localStorage (jika ada)
            let cart = JSON.parse(localStorage.getItem('cart')) || [];

            // Tambahkan produk ke keranjang
            cart.push(product);

            // Simpan kembali ke localStorage
            localStorage.setItem('cart', JSON.stringify(cart));

            // Arahkan ke halaman cart
            window.location.href = 'cart.html';

        function loadCart() {
            const cart = JSON.parse(localStorage.getItem('cart')) || [];
            const cartItemsDiv = document.getElementById('cart-items');
            const totalPriceSpan = document.getElementById('total-price');
            let total = 0;

            if (cart.length === 0) {

                    <p>Harga: Rp ${item.price.toLocaleString('id-ID')}</p>
                    <hr>
                `;
                cartItemsDiv.appendChild(div);
                total += item.price;
            });

            totalPriceSpan.textContent = total.toLocaleString('id-ID');
        }
